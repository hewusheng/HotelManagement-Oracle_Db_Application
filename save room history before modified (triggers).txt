--先创建这个表，房间历史信息

CREATE TABLE  GP_ROOM_HISTORY
   (	ROOM_ID NUMBER(25,0) NOT NULL ENABLE, 
	SQUAREFEET NUMBER(25,0) NOT NULL ENABLE, 
	ROOMTYPE VARCHAR2(25) NOT NULL ENABLE, 
	ROOMDESCRIPTION VARCHAR2(25) NOT NULL ENABLE
   ) ;

--然后创建这个trigger，保存房间历史

CREATE OR REPLACE TRIGGER  GP_ROOMINVENTORY_TRG1
BEFORE
update on GP_ROOMINVENTORY
for each row
begin
 INSERT INTO gp_room_history
 VALUES (:OLD.ROOM_ID, :OLD.SQUAREFEET, :OLD.ROOMTYPE, :OLD.ROOMDESCRIPTION);

end;
